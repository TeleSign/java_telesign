010package com.telesign.phoneid;
011
012import com.google.gson.Gson;
013import com.telesign.phoneid.response.PhoneIdContactResponse;
014import com.telesign.phoneid.response.PhoneIdLiveResponse;
015import com.telesign.phoneid.response.PhoneIdScoreResponse;
016import com.telesign.phoneid.response.PhoneIdStandardResponse;
017import com.telesign.util.PhoneUtil;
018import com.telesign.util.TeleSignRequest;
019import java.io.IOException;
020
021/**
022 *  The PhoneId class abstracts your interactions with the <em>TeleSign PhoneID web service</em>.
023 *  A PhoneId object encapsulates your credentials (your TeleSign <em>Customer ID</em> and <em>Secret Key</em>).
024 */
025public class PhoneId
026{
027        /**
028         * The PhoneId class constructor.
029         * Once you instantiate a PhoneId object, you can use it to make instance calls to <em>PhoneID Standard</em>, <em>PhoneID Score</em>, <em>PhoneID Contact</em>, and <em>PhoneID Live</em>.
030         * @param customer_id   [Required] A string representing your TeleSign Customer ID. This represents your TeleSign account number.
031         * @param secret_key    [Required] A string representing your TeleSign Secret Key (available from the TeleSign Client Portal).
032         */
033        public PhoneId(String customer_id, String secret_key)
034        {
035            this.customer_id = customer_id;
036            this.secret_key = secret_key;
037        }
038        
039        /**
040         * Returns information about a specified phone number’s type, numbering structure, cleansing details, and location details. 
041         * @param identifier    [Required] A string representing the phone number you want information about.
042         * @return              A {@link com.telesign.phoneid.response.PhoneIdStandardResponse} object, which contains the JSON-formatted response body from the TeleSign server.
043         */
044        public PhoneIdStandardResponse standard(String identifier)
045        {
046                String result = null;
047                
048                try
049                {
050                        String clean_phone = PhoneUtil.formatTo11Digits(identifier);
051                        TeleSignRequest tr = new TeleSignRequest("https://rest.telesign.com", "/v1/phoneid/standard/" + clean_phone, "GET", customer_id, secret_key);
052                        result = tr.executeRequest();
053                }
054                catch (IOException e)
055                {
056                        System.err.println("IOException while executing phoneid API: " + e.getMessage());
057                }
058                
059                Gson gson = new Gson();
060                PhoneIdStandardResponse response = gson.fromJson(result, PhoneIdStandardResponse.class);
061                
062                return response;
063        }
064        
065        /**
066         * Returns risk information about a specified phone number, including a real-time risk score, threat level, and recommendation for action. 
067         * @param identifier    [Required] A string representing the phone number you want information about.
068         * @param ucid          [Required] A string specifying one of the Use Case Codes.
069         * @return              A {@link com.telesign.phoneid.response.PhoneIdScoreResponse} object, which contains the JSON-formatted response body from the TeleSign server.
070         */
071        public PhoneIdScoreResponse score(String identifier, String ucid)
072        {
073                String result = null;
074                
075                try
076                {
077                        String clean_phone = PhoneUtil.formatTo11Digits(identifier);
078                        TeleSignRequest tr = new TeleSignRequest("https://rest.telesign.com", "/v1/phoneid/score/" + clean_phone, "GET", customer_id, secret_key);
079                        tr.addParam("ucid", ucid);      
080                        result = tr.executeRequest();
081                }
082                catch (IOException e)
083                {
084                        System.err.println("IOException while executing phoneid API: " + e.getMessage());
085                }
086                
087                Gson gson = new Gson();
088                PhoneIdScoreResponse response = gson.fromJson(result, PhoneIdScoreResponse.class);
089                
090                return response;
091        }
092        
093        /**
094         * Returns contact details for a specified phone number’s subscriber. This includes the subscriber's First Name, Last Name, Street Address, City, State (or Province), Country, and ZIP (Postal) Code.
095         * @param identifier    [Required] A string representing the phone number you want information about.
096         * @param ucid          [Required] A string specifying one of the Use Case Codes.
097         * @return              A {@link com.telesign.phoneid.response.PhoneIdContactResponse} object, which contains the JSON-formatted response body from the TeleSign server.
098         */
099        public PhoneIdContactResponse contact(String identifier, String ucid)
100        {
101                String result = null;
102                
103                try
104                {
105                        String clean_phone = PhoneUtil.formatTo11Digits(identifier);
106                        TeleSignRequest tr = new TeleSignRequest("https://rest.telesign.com", "/v1/phoneid/contact/" + clean_phone, "GET", customer_id, secret_key);
107                        tr.addParam("ucid", ucid);
108                        
109                        result = tr.executeRequest();                   
110                }
111                catch (IOException e)
112                {
113                        System.err.println("IOException while executing phoneid API: " + e.getMessage());
114                }
115                
116                Gson gson = new Gson();
117                PhoneIdContactResponse response = gson.fromJson(result, PhoneIdContactResponse.class);
118                
119                return response;
120        }
121        
122        /**
123         *  Returns information about a specified phone number’s <em>state of operation</em>.
124         *  You can use it to find out 
125         *  if the line is in service, 
126         *  if the number is reachable, 
127         *  if the mobile phone is roaming, and if so, in which country.
128         * @param identifier    [Required] A string representing the phone number you want information about.
129         * @param ucid          [Required] A string specifying one of the Use Case Codes.
130         * @return              A {@link com.telesign.phoneid.response.PhoneIdContactResponse} object, which contains the JSON-formatted response body from the TeleSign server.
131         */
132        public PhoneIdLiveResponse live(String identifier, String ucid)
133        {
134                String result = null;
135                
136                try
137                {
138                        String clean_phone = PhoneUtil.formatTo11Digits(identifier);
139                        TeleSignRequest tr = new TeleSignRequest("https://rest.telesign.com", "/v1/phoneid/live/" + clean_phone, "GET", customer_id, secret_key);
140                        tr.addParam("ucid", ucid);      
141                        result = tr.executeRequest();
142                }
143                catch (IOException e)
144                {
145                        System.err.println("IOException while executing phoneid API: " + e.getMessage());
146                }
147                
148                Gson gson = new Gson();
149                PhoneIdLiveResponse response = gson.fromJson(result, PhoneIdLiveResponse.class);
150                
151                return response;
152        }
153        
154        private final String customer_id;
155    private final String secret_key;
156}
