010package com.telesign.phoneid.response;
011
012import com.google.gson.Gson;
013
014/** The <strong>PhoneIdScoreResponse</strong> class encapsulates all of the information returned from a call to the <strong>PhoneID Score</strong> web service.  */
015public class PhoneIdScoreResponse {
016        /** A String containing a <em>reference identifier</em> that uniquely identifies the Request message that initiated this Response. */
017        public String reference_id;
018        
019        /** A String containing the URI for accesses the PhoneID resource. */
020        public String resource_uri;
021
022        /** A String containing the name of the subresource that was accessed. For example, "standard". */
023        public String sub_resource;
024        
025        /** An array of {@link Error} objects. Each Error object contains information about an error condition that might have resulted from the Request. */    
026        public Error[] errors;
027
028        /** A String containing the Signature, exactly as it was sent in the Request. Unless the request contained an invalid Signature, this contains an empty string.*/
029        public String signature_string;
030
031        /** An object containing details about the request status. */
032        public Status status;
033
034        /** An object containing details about the numbering attributes of the specified phone number. */
035        public Numbering numbering;
036
037        /** An object that describes the risk score for the phone number specified in the request. */
038        public Risk risk;
039        
040        /** An array of property-value pairs, that contain information on error conditions that might have resulted from the Request. */
041        public static class Error
042        {
043                /**  A 1 to 5-digit error code (possibly negative) that indicates the type of error that occurred. When no error occurs, the default value 0 is returned. */
044                public int code;
045                
046                /** A string that describes the type of error that occurred. If no error occurs, this parameter is empty. */
047                public String description;
048        }
049
050        /**  An object containing details about the phone type. */
051        public static class PhoneType {
052            
053                /** One of the <emphasis>Phone Type Codes</emphasis>." */
054                public int code;
055                
056                /**  A description of the phone type. */
057                public String description;
058        }
059
060        /** An object containing details about the request status. */
061        public static class Status {
062                
063                /** An ISO 8601 UTC timestamp indicating when the transaction status was updated. */
064                public String updated_on;
065                
066                /**  One of the<emphasis> Transaction Status Codes</emphasis>. */
067                public int code;
068                
069                /** A descriptionm of the transaction status. */
070                public String description;
071        }
072
073        /** An object containing details about the numbering attributes of the specified phone number. */
074        public static class Numbering {
075                
076                /** An object containing details about the original phone number passed to TeleSign’s PhoneID API. */
077                public OriginalNumber original;
078                
079                /** An object containing details about how the phone number was cleansed. Phone Cleansing corrects common formatting issues in submitted phone numbers. */
080                public CleansingNumber cleansing;
081                
082                /** An object containing details about the original phone number passed to TeleSign’s PhoneID API. */
083                public static class OriginalNumber {
084                    
085                        /** The Base Phone Number. This is simply the phone number without the Country Dialing Code. */
086                        public String phone_number;
087                        
088                        /** The Base Phone Number prefixed with the Country Dialing Code. This forms the Subresource Identifier part of the PhoneID Contact web service URI. */
089                        public String complete_phone_number;
090                        
091                        /**  A 1, 2, or 3-digit number representing the Country Dialing Code. For example, the Country Dialing Code for both the U.S.A. and Canada is 1, and the Country Dialing Code for the United Kingdom is 44. */
092                        public String country_code;
093                }
094        
095                /** An object containing details about how the phone number was cleansed. Phone Cleansing corrects common formatting issues in submitted phone numbers. */
096                public static class CleansingNumber {
097                    
098                        /** An object containing cleansing details about a phone number used for receiving text messages. */    
099                        public Number sms;
100                        
101                        /** An object containing cleansing details about a phone number used for receiving voice calls. */
102                        public Number call;
103                        
104                        /** An object containing cleansing details about a phone number. */
105                        public static class Number {
106                            
107                                /** The Base Phone Number. This is simply the phone number without the Country Dialing Code. */
108                                public String phone_number;
109                                
110                                /**  A 1, 2, or 3-digit number representing the Country Dialing Code. For example, the Country Dialing Code for both the U.S.A. and Canada is 1, and the Country Dialing Code for the United Kingdom is 44. */
111                                public String country_code;
112                                
113                                /**  The minimum number of digits allowed for phone numbers with this particular Country Dialing Code. */
114                                public int min_length;
115                                
116                                /** The maximum number of digits allowed for phone numbers with this particular Country Dialing Code. */
117                                public int max_length;
118                                
119                                /** One of the Phone Number Cleansing Codes describing the cleansing operation TeleSign performed on the phone number. The default value is 100 (No changes were made to the phone number). */
120                                public int cleansed_code;
121                        }
122                }
123        }
124
125        /** An object that describes the risk score for the phone number specified in the request. */
126        public static class Risk {
127            
128                /** A string indicating the severity of the risk. */
129                public String level;
130                
131                /** One of the <strong>Score</strong> values. */
132                public int score;
133                
134                /** A string indicating the action that TeleSign recommends that you take based on the risk score. */
135                public String recommendation;
136        }
137
138        /**
139         * Converts a <strong>PhoneIdScoreResponse</strong> object to its equivalent JSON format.  
140         * @see java.lang.Object#toString()
141         * @return A string containing the response from the TeleSign server, in JSON format.  
142         */
143        @Override
144        public String toString() {
145                Gson gson = new Gson();
146                return gson.toJson(this);
147        }
148}
