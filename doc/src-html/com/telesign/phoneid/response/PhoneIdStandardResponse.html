010package com.telesign.phoneid.response;
011
012import com.google.gson.Gson;
013
014/** The <strong>PhoneIdStandardResponse</strong> class encapsulates all of the information returned from a call to the <strong>PhoneID Standard</strong> web service.  */
015public class PhoneIdStandardResponse
016{
017        /** A String containing a <em>reference identifier</em> that uniquely identifies the Request message that initiated this Response. */
018        public String reference_id;
019        
020        /** A String containing the URI for accesses the PhoneID resource. */
021        public String resource_uri;
022
023        /** A String containing the name of the subresource that was accessed. For example, "standard". */
024        public String sub_resource;
025        
026        /** An array of {@link Error} objects. Each Error object contains information about an error condition that might have resulted from the Request. */    
027        public Error[] errors;
028
029        /** An object that contains details about the phone type. */
030        public PhoneType phone_type;
031
032        /** A String containing the Signature, exactly as it was sent in the Request. Unless the request contained an invalid Signature, this contains an empty string.*/
033        public String signature_string;
034
035        /** An object containing details about the request status. */
036        public Status status;
037
038        /** An object containing details about the numbering attributes of the specified phone number. */
039        public Numbering numbering;
040
041        /** An object containing geographical location information associated with the phone number. */
042        public Location location; 
043
044        /** An array of property-value pairs, that contain information on error conditions that might have resulted from the Request. */
045        public static class Error
046        {
047                /**  A 1 to 5-digit error code (possibly negative) that indicates the type of error that occurred. When no error occurs, the default value 0 is returned. */
048                public int code;
049                
050                /** A string that describes the type of error that occurred. If no error occurs, this parameter is empty. */
051                public String description;
052        }
053
054        /**  An object containing details about the phone type. */
055        public static class PhoneType {
056            
057                /** One of the <emphasis>Phone Type Codes</emphasis>." */
058                public int code;
059                
060                /**  A description of the phone type. */
061                public String description;
062        }
063
064        /** An object containing details about the request status. */
065        public static class Status {
066                
067                /** An ISO 8601 UTC timestamp indicating when the transaction status was updated. */
068                public String updated_on;
069                
070                /**  One of the<emphasis> Transaction Status Codes</emphasis>. */
071                public int code;
072                
073                /** A descriptionm of the transaction status. */
074                public String description;
075        }
076
077        /** An object containing details about the numbering attributes of the specified phone number. */
078        public static class Numbering {
079                
080                /** An object containing details about the original phone number passed to TeleSign’s PhoneID API. */
081                public OriginalNumber original;
082                
083                /** An object containing details about how the phone number was cleansed. Phone Cleansing corrects common formatting issues in submitted phone numbers. */
084                public CleansingNumber cleansing;
085                
086                /** An object containing details about the original phone number passed to TeleSign’s PhoneID API. */
087                public static class OriginalNumber {
088                    
089                        /** The Base Phone Number. This is simply the phone number without the Country Dialing Code. */
090                        public String phone_number;
091                        
092                        /** The Base Phone Number prefixed with the Country Dialing Code. This forms the Subresource Identifier part of the PhoneID Contact web service URI. */
093                        public String complete_phone_number;
094                        
095                        /**  A 1, 2, or 3-digit number representing the Country Dialing Code. For example, the Country Dialing Code for both the U.S.A. and Canada is 1, and the Country Dialing Code for the United Kingdom is 44. */
096                        public String country_code;
097                }
098        
099                /** An object containing details about how the phone number was cleansed. Phone Cleansing corrects common formatting issues in submitted phone numbers. */
100                public static class CleansingNumber {
101                    
102                        /** An object containing cleansing details about a phone number used for receiving text messages. */    
103                        public Number sms;
104                        
105                        /** An object containing cleansing details about a phone number used for receiving voice calls. */
106                        public Number call;
107                        
108                        /** An object containing cleansing details about a phone number. */
109                        public static class Number {
110                            
111                                /** The Base Phone Number. This is simply the phone number without the Country Dialing Code. */
112                                public String phone_number;
113                                
114                                /**  A 1, 2, or 3-digit number representing the Country Dialing Code. For example, the Country Dialing Code for both the U.S.A. and Canada is 1, and the Country Dialing Code for the United Kingdom is 44. */
115                                public String country_code;
116                                
117                                /**  The minimum number of digits allowed for phone numbers with this particular Country Dialing Code. */
118                                public int min_length;
119                                
120                                /** The maximum number of digits allowed for phone numbers with this particular Country Dialing Code. */
121                                public int max_length;
122                                
123                                /** One of the Phone Number Cleansing Codes describing the cleansing operation TeleSign performed on the phone number. The default value is 100 (No changes were made to the phone number). */
124                                public int cleansed_code;
125                        }
126                }
127        }
128
129        /**
130         * An object containing geographical location information associated with the phone number.
131         * <strong>Note:</strong> <emphasis>Location</emphasis> refers to the place where the phone is registered, <emphasis>not</emphasis> to where it actually is at the time of this request. 
132         */
133        public static class Location {
134            
135                /** A string specifying the name of the County (or Parish) associated with the phone number (U.S. only). */
136                public String county;
137                
138                /** A string specifying the name of the city associated with the phone number. */
139                public String city;
140                
141                /** The 2-letter State Code of the state (province, district, or territory) associated with the phone number (North America only). */
142                public String state;
143                
144                /** The 5-digit United States Postal Service ZIP Code associated with the phone number (U.S. only). */
145                public String zip;
146                
147                /** An object containing details about the country associated with the phone number. */
148                public Country country;
149                
150                /** An object containing details about the Time Zone associated with the phone number. */
151                public TimeZone time_zone;
152                
153                /** An object containing details about the geographical coordinates of the location where the phone number is registered. */
154                public Coordinates coordinates;
155                
156                /** A 4-digit string indicating the Primary Metropolitan Statistical Area (PMSA) Code for the location associated with the phone number (U.S. only). PMSA Codes are governed by the US Census Bureau. */
157                public String metro_code;
158                
159                /** An object containing details about the country associated with the phone number. */
160                public static class Country {
161                    
162                        /** The ISO 3166-1 2-letter Country Code associated with the phone number. */
163                        public String iso2;
164                        
165                        /** The ISO 3166-1 3-letter Country Code associated with the phone number */
166                        public String iso3;
167                        
168                        /** The Country Name associated with phone number. */
169                        public String name;
170                }
171
172                /** An object containing details about the Time Zone associated with the phone number. */
173                public static class TimeZone {
174                    
175                        /** For U.S. domestic phone numbers, this parameter returns the UTC offset associated with the phone number. */
176                        public String utc_offset_min;
177                        
178                        /** A string identifying the Time Zone Name (TZ) associated with the phone number (U.S. only). For example: "America/Los_Angeles". */
179                        public String name;
180                        
181                        /** For international phone numbers, this parameter returns the maximum UTC offset for the country associated with the phone number. For U.S. domestic phone numbers, this parameter returns the same result as utc_offset_min. */
182                        public String utc_offset_max;
183                }
184
185                /** An object containing details about the geographical coordinates of the location where the phone number is registered. */
186                public static class Coordinates {
187                    
188                        /** A value indicating the number of degrees of latitude of the location associated with the phone number, expressed in seven decimal digits, with five decimal places. For example, 34.18264. */
189                        public double latitude;
190                        
191                        /** A value indicating the number of degrees of longitude of the location associated with the phone number, expressed in eight decimal digits, with five decimal places For example, -118.30840. */
192                        public double longitude;
193                }
194        }
195
196        /**
197         * Converts a <strong>PhoneIdStandardResponse</strong> object to its equivalent JSON format.  
198         * @see java.lang.Object#toString()
199         * @return A string containing the response from the TeleSign server, in JSON format.  
200         */
201        @Override
202        public String toString()
203        {
204                Gson gson = new Gson();
205                return gson.toJson(this);
206        }
207}
