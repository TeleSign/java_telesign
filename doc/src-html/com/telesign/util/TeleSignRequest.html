031/**
032 * The TeleSignRequest class is an abstraction for creating and sending HTTP 1.1 REST Requests for TeleSign web services.  
033 */
034public class TeleSignRequest
035{
036        /** A boolean value indicates the Request Method. <em>True</em> for <strong>POST</strong>, and <em>False</em> for <strong>GET</strong>. */
037        private boolean post;
038                
039        /** The Base URI. For TeleSign web services, this is https://rest.telesign.com/. */
040        private final String base;
041        
042        /** The name of the network resource. Each of the TeleSign web services is identified by its resource specifier. */
043        private final String resource;
044        
045        /** Your TeleSign Customer ID. This represents your TeleSign account number. */
046        private final String customer_id;
047        
048        /** Your TeleSign Secret Shared Key (available from the TeleSign Client Portal). */
049        private final String secret_key;
050
051        /** The set of HTTP 1.1 Request header field/value pairs. */
052        private TreeMap<String, String> headers;
053        
054        /** The set of <em>TeleSign-specific</em> Request header field/value pairs. */
055        private TreeMap<String, String> ts_headers;       
056        
057        /** The set of HTTP 1.1 Request parameter attribute/value pairs. */
058        private HashMap<String, String> params;
059
060        /** The <em>absolute form</em> of the Resource URI */
061        private URL url;
062        
063        /** The persistent session connection for the Request. */
064        private HttpURLConnection connection;
065        
066        /** The contents of the entity body in the POST Request. */
067        private String body = "";
068        
069        /** A boolean value that indicates whether the <em>TeleSign-specific</em> <strong>Date</strong> Request header field is used. */
070        private boolean ts_date = false;
071
072        /**
073         * The TeleSitgnRequest class constructor.
074         * A TeleSitgnRequest object contains all of the information required to call any/all of the TeleSign web services. 
075         * @param base          [Required] A string representing the Base URI. For TeleSign web services, this is https://rest.telesign.com/. 
076         * @param resource              [Required] A string representing the name of the network resource. Each of the TeleSign web services is identified by its resource specifier.
077         * @param method                [Required] A string representing the method to be performed on the resource. For TeleSign web services, this is either GET or POST. 
078         * @param customer_id   [Required] A string representing your TeleSign Customer ID. This represents your TeleSign account number.
079         * @param secret_key    [Required] A string representing your TeleSign Secret Shared Key (available from the TeleSign Client Portal).
080         */
081        public TeleSignRequest(String base, String resource, String method, String customer_id, String secret_key)
082        {
083                post = (method.toLowerCase().equals("post"));
084                
085                this.base                       = base;
086                this.resource           = resource;
087                this.customer_id        = customer_id;
088                this.secret_key         = secret_key;
089                
090                ts_headers              = new TreeMap<String, String>();
091                headers         = new TreeMap<String, String>();
092                params          = new HashMap<String, String>();
093        }
094        
095        /**
096         * Adds an HTTP 1.1 Request header field/value pair to the set of Request Headers.
097         * @param key           [Required] A string representing a Request header field <em>name</em>. 
098         * @param value [Required] A string representing a Request header field <em>value</em>.
099         */
100        public void addHeader(String key, String value)
101        {
102                // Check to see if this request header field is a "TeleSign-specific" header field.  
103                if((key.length() > 4) && (key.toLowerCase().substring(0, 5).equals("x-ts-")))
104                {                       
105                        // If using the TeleSign-specific date header, then use a blank line for the standard Date Request header.  
106                        if(key.toLowerCase().equals("x-ts-date"))
107                                ts_date = true;
108                        
109                        ts_headers.put(key, value);
110                }
111                
112                headers.put(key, value);
113        }
114        
115        /**
116         * Adds an HTTP 1.1 Request parameter attribute/value pair to the set of Request Parameters.
117         * @param key           [Required] A string representing a Request parameter attribute <em>name</em>.
118         * @param value [Required] A string representing a Request parameter attribute <em>value</em>.
119         */
120        public void addParam(String key, String value)
121        {
122                params.put(key, value);
123        }
124        
125        /**
126         * Sets the entity body of an HTTP 1.1 POST Request message. Contains the Request parameters for <strong>Verify.call</strong> and <strong>Verify.sms</strong>.
127         * @param post_body     [Required] A string that contains the contents of the POST Request's entity body.
128         */
129        public void setPostBody(String post_body)
130        {
131                body = post_body;
132        }
133        
134        /**
135         * Gets the entity body of an HTTP 1.1 POST Request message. Used for calls to <strong>Verify.call</strong> and <strong>Verify.sms</strong>.
136         * @return      A string that contains the contents of the POST Request's entity body.
137         */
138        public String getPostBody()
139        {
140                return body;
141        }
142        
143        /**
144         * Returns the set of <em>TeleSign-specific</em> Request header fields.    
145         * @return      A sorted key/value mapping that contains the TeleSign-specific Request header fields.
146         */
147        public Map<String, String> getTsHeaders()
148        {
149                return ts_headers;
150        }
151        
152        /**
153         * Returns the set of both the standard <u>and</u> the <em>TeleSign-specific</em> Request header fields.
154         * @return      A sorted key/value mapping that contains all of the Request header fields.
155         */
156        public Map<String, String> getAllHeaders()
157        {
158                TreeMap<String, String> tmp = new TreeMap<String, String>();
159                
160                tmp.putAll(headers);
161                tmp.putAll(ts_headers);
162                
163                return tmp;
164        }
165
166        /**
167         * Returns the set of POST Request parameters.
168         * @return      A sorted key/value mapping that contains all of the POST Request parameters.
169         */
170        public Map<String, String> getAllParams()
171        {
172                return params;
173        }
174        
175        /**
176         * Creates and sends the REST request.
177         * @return                      A string containing the TeleSign web server's Response.
178         * @throws IOException  A {@link java.security.SignatureException} signals that an error occurred while attempting to sign the Request.
179         */
180        public String executeRequest() throws IOException
181        {
182                String signingString = getSigningString(customer_id); 
183                String signature;
184                String url_output = "";
185                
186                // Create the absolute form of the resource URI, and place it in a string buffer. 
187                StringBuffer full_url = new StringBuffer(base).append(resource);
188                
189                if(params.size() > 0)
190                {
191                        full_url.append("?"); int i =0;         // The first parameter is always prefixed with a question mark.
192                        
193                        for(String key : params.keySet())
194                        {
195                                if(++i != 0)                            // Additonal parameters are always prefixed with an ampersands.
196                                        full_url.append("&"); 
197                                
198                                full_url.append(URLEncoder.encode(key, "UTF-8")).append("=").append(URLEncoder.encode(params.get(key), "UTF-8"));
199                        }
200                }
201                
202                /* Convert the URL string into a URL object,
203                 * to use later for creating a 
204                 * persistent connection for the session.
205                 */
206                url = new URL(full_url.toString());             
207                
208                /* Create the Signature using the formula:
209                 * Signature = Base64( HMAC-SHA1( YourTeleSignAPIKey, UTF-8-Encoding-Of( StringToSign ) ) 
210                 */
211                try
212                {
213                        signature = hmacsha1(signingString, secret_key);
214                }
215                catch (SignatureException e)
216                {
217                        System.err.println("Error signing request " + e.getMessage());
218                        
219                        return null;
220                }
221                
222                /* Create the Authorization header using the following formula:
223                 * Authorization = "TSA" + " " + Customer ID + ":" + Signature.
224                 */
225                String auth_header = "TSA" + customer_id + ":" + signature;
226                
227                /* Create an HTTP Connection object to
228                 * open a persistent connection to the URL, 
229                 * and begin a new session for the Request/Response transaction. 
230                 */
231                connection = (HttpURLConnection) url.openConnection();          
232                connection.setConnectTimeout(30000);
233
234                // Send the Request message headers to the web resource.
235                connection.setRequestProperty("Authorization", auth_header);    // Send the Authorization header.
236                
237                if(post)                                                                                                                        // This Request message contains an enitiy body, so
238                        connection.setRequestProperty("Content-Length", Integer.toString(body.length()));       // Send the size of the message payload as
239                                                                                                                                                // the value of the Content-Length header.  
240                for(String key : ts_headers.keySet())
241                        connection.setRequestProperty(key, ts_headers.get(key));        // Add the TeleSign-specific Request headers.
242
243                for(String key : headers.keySet())
244                        connection.setRequestProperty(key, headers.get(key));           // Add the generl Request headers.
245                
246                if(post)                                                                        // This Request message contains an enitiy body, so
247                {                                                                               // use the connection for outputting (writing a stream of) information,
248                    connection.setDoOutput(true);                               // as opposed to false, for inputting.
249                    
250                    DataOutputStream wr = new DataOutputStream (connection.getOutputStream ());
251                    
252                    wr.writeBytes(body);                                                // Write the entity body for the POST Request.
253                    wr.flush(); wr.close();
254                }
255                
256                // Sending Request now complete. 
257                
258                
259                // Retrieve the Status Code from the web server.                   The Status-Code is a 3-digit integer that indicates the result
260                int response = connection.getResponseCode();            // of the web server's attempt to understand and satisfy your request.          
261                
262                // Retrieve the contents of the Response message from the TeleSign server.
263                BufferedReader in;
264                
265                try
266                {
267                    // Extract the the contents of the Response message with a stream object.
268                    InputStream  isr = (response == 200) ? connection.getInputStream() : connection.getErrorStream();   // When successful, the value is "200".
269                        
270                        in = new BufferedReader(new InputStreamReader(isr));    // Read the input stream into a buffer.
271                        
272                        String urlReturn;                                               // Transfer the contents of the buffer to string variable.
273        
274                        while ((urlReturn = in.readLine()) != null)             // The response headers are separated using the new line character ("\n").
275                            url_output += urlReturn;                                    // Information returned in the response body is structured in JSON format.
276                        
277                        in.close();                     
278                }
279                catch (IOException e)
280                {
281                        System.err.println("IOException while reading from input stream " + e.getMessage());
282                }               
283                
284                return url_output;
285        }

287        /**
288         * Create the <em>string to sign</em>.
289         * This is a <em>helper method</em>, used interanlly by the {@link com.telesign.util.TeleSignRequest#executeRequest()} method.
290         * @param customer_id   [Required] A string representing your TeleSign Customer ID. This represents your TeleSign account number.
291         * @return                                      A string representing the signing string used to create a Signature object, for the authenticate the REST request.  
292         */
293        private String getSigningString(String customer_id)
294        {
295                // Create the stringToSign element of the Signature component.
296                String stringToSign = post ? "POST\n": "GET\n";
297                String content_type = post ? "application/x-www-form-urlencoded\n" : "\n";
298
299                // Concatonate the Request headers to the stringToSign. 
300                stringToSign += content_type;
301                
302                /* Add the Date header to both the set of Headers,
303                /* and to the stringToSign (killing two birds with one stone).
304                 */
305                if(ts_date)                                     // If you use the TeleSign-specific Date header, 
306                        stringToSign += "\n";           // then the standard Date header isn't required,
307                else                                            // so just insert an empty line into the StringToSign (as a placeholder).
308                {
309                        // Create a standard HTTP 1.1 Date header field.
310                        Date now = new Date();
311                        
312                        /* Convert the Date object to conform to the RFC 2616, Section 3.3.1 format.
313                         * For example : Tue, 10 Jan 2012 19:36:42 +0000. 
314                         */
315                        SimpleDateFormat rfc2616 = new SimpleDateFormat("EEE, dd MMM YYYY HH:mm:ss ZZZZ");                      
316                        String date = rfc2616.format(now);
317                        addHeader("Date", date);
318                        
319                        stringToSign += date + "\n";
320                }
321                
322                /* Create the CanonicalizedTsHeaders element. This entails:
323                 * Converting header names to lowercase,
324                 * Sorting the set by header name,
325                 * Ensuring that each X-TS- header name appears just once,
326                 * Unfolding long header values that span multiple lines, by replacing the folding white space (newline) with a single space, and
327                 * Removing white space surrounding the colon that appears between each header name/value pair.
328                 */
329                Set<String> orderedKeys = ts_headers.keySet();
330                
331                for(String key : orderedKeys)                                                                   // Add each of the X-TS- headers (name:value pairs) to the stringToSign,
332                        stringToSign += key.toLowerCase() + ":" + ts_headers.get(key) + "\n";   // each one separated by a newline.
333                
334                // Create the CanonicalizedPOSTVariables element.
335                if(post)
336                        stringToSign += body + "\n";                    // Just concatonate the POST Request entity body "as-is" to the stringToSign.
337                
338                // Create the CanonicalizedResource element. 
339                stringToSign += resource;                                       // Just concatonate the resource "as-is" to the stringToSign.
340                
341                return stringToSign;
342        }

344        /**
345         * Creates the Signature component for the Authorization header field. Uses your Secret Key to encode the stringToSign.   
346         * This is a <em>helper method</em>, used interanlly by the {@link TeleSignRequest#executeRequest()} method.
347         * @param data  [Required] The stringToSign.  
348         * @param key           [Required] Your TeleSign API Key. Also known as your Secret Key, and your Shared Cryptographic Key. It s a bese64-encoded string value.
349         * @return              A String containing the Base64-encoded hash-based message authentication code.
350         * @throws java.security.SignatureException     Failed to generate HMAC. IllegalArgumentException - if algorithm is null or key is null or empty.
351         */
352        private String hmacsha1(String data, String key)  throws java.security.SignatureException
353        {
354                String HMAC_SHA1_ALGORITHM = "HmacSHA1";
355                String result;
356                
357                byte[] decoded_key = Base64.decodeBase64(key);  // Get the raw key value.
358                
359                try
360                {
361                        /* Construct a another secret key from the raw key, 
362                         * one that can be used with the specified cryptographic hash algorithm. 
363                         */
364                        SecretKeySpec signingKey = new SecretKeySpec(decoded_key, HMAC_SHA1_ALGORITHM);
365                        
366                        // Create a MAC object, which implements the SHA-1 hash algorithm.
367                        Mac mac = Mac.getInstance(HMAC_SHA1_ALGORITHM);
368                        mac.init(signingKey);                                                           //  Use the new encryption key to for encrypting the stringToSign.
369                        
370                        // Encrypt the stringToSign.
371                        byte[] rawHmac = mac.doFinal(data.getBytes());                  // This renders the encrypted for of the stringToSign as chunk of binary data.
372                        
373                        // Encode the binary data into an ASCII string format - so it can be sent reliably over the Internet (a medium originally designed for transporting textual data). 
374                        result =  new String(Base64.encodeBase64(rawHmac));
375                        
376                }
377                catch (Exception e)
378                {
379                        throw new SignatureException("Failed to generate HMAC : " + e.getMessage());
380                }
381                
382                return result;
383        }
