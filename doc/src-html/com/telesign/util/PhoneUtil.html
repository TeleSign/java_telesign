010package com.telesign.util;
011
012/**
013 * The PhoneUtil class contains a mentod for normalizing phone numbers.
014 */
015public class PhoneUtil
016{
017        /**
018         * This mentod is a helper method for cleansing phone numbers. 
019         * Specifically, it ensures that all phone number conform to a standard 11 digit format.
020         * For example, (310) 555-1212 becomes 13105551212.
021         * @param phoneNumber   [Required] A string that contains the user's phone number, in its original form. 
022         * @return                                      The cleansed form of the phone number.
023         */
024        public static String formatTo11Digits(String phoneNumber)
025        {
026                if(phoneNumber == null) return null;
027
028                // Remove non-numeric characters (leaving digits 0..9). 
029                String cleanPhone = phoneNumber.replaceAll("[^\\d]", "");
030
031                // Assert that the characters in the cleaned phone number are all numeric.
032                try
033                {
034                        Long.parseLong(cleanPhone);             // Convert the string representation of the cleaned phone number to a long value. 
035                }
036                catch(NumberFormatException nfe)
037                {
038                        return null;
039                }
040
041                switch(cleanPhone.length())
042                {
043                        case 0: case 1: case 2: case 3: case 4:
044                        case 5: case 6: case 7: case 8: case 9:
045                                return null;
046                        case 10:                                                // Missing the dialing code for North America (+1).
047                                return "1" + cleanPhone;
048                        case 11:
049                                return cleanPhone;
050                        case 12: 
051                                // If the phone number is 12 digits long, and begins with either "11" or "01", then treat it as country code and omit.
052                                // For example 113102669619
053                                if(cleanPhone.charAt(0) == '0')
054                                {
055                                        return cleanPhone.substring(1);         // Remove the leading zero.
056                                } 
057                                return null;
058                        case 13:
059                                if(cleanPhone.startsWith("00"))
060                                {
061                                        return cleanPhone.substring(2);         // Remove the leading two zeros.
062                                }
063                                return null;
064                        default:
065                                return null;
066                }
067        }
068}
