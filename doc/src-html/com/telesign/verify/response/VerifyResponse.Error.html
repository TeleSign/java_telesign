010package com.telesign.verify.response;
011
012import com.google.gson.Gson;
013
014/** The <strong>VerifyResponse</strong> class encapsulates all of the information returned from a call to either the <strong>Verify SMS</strong> web service, or to the <strong>Verify Call</strong>  web service. */
015public class VerifyResponse {
016        /** A String containing a <em>reference identifier</em> that uniquely identifies the Request message that initiated this Response. */
017        public String reference_id;
018        
019        /** A String containing the URI for accesses the PhoneID resource. */
020        public String resource_uri;
021
022        /** A String containing the name of the subresource that was accessed. For example, "standard". */
023        public String sub_resource;
024        
025        /** An array of {@link Error} objects. Each Error object contains information about an error condition that might have resulted from the Request. */    
026        public Error[] errors;
027
028        /** An object containing details about the request status. */
029        public Status status;
030
031        /** An object that describes the verification status. */
032        public Verify verify;
033        
034        /** An array of property-value pairs, that contain information on error conditions that might have resulted from the Request. */
035        public static class Error
036        {
037                /**  A 1 to 5-digit error code (possibly negative) that indicates the type of error that occurred. When no error occurs, the default value 0 is returned. */
038                public int code;
039                
040                /** A string that describes the type of error that occurred. If no error occurs, this parameter is empty. */
041                public String description;
042        }
043
044        /** An object containing details about the request status. */
045        public static class Status {
046                
047                /** An ISO 8601 UTC timestamp indicating when the transaction status was updated. */
048                public String updated_on;
049                
050                /**  One of the<emphasis> Transaction Status Codes</emphasis>. */
051                public int code;
052                
053                /** A descriptionm of the transaction status. */
054                public String description;
055        }
056
057        /** An object that describes the verification status. */
058        public static class Verify {
059            
060                /** Indicates whether the verification code entered matches that which was sent. Possible values are VALID, INVALID, or UNKNOWN. When the code entered matches the code sent, the response will be VALID. When the code entered does not match the code sent, code_state will be INVALID. */
061                public String code_state;
062                
063                /**  Always set to an empty string. */
064                public String code_entered;
065        }
066
067        /**
068         * Converts a <strong>VerifyResponse</strong> object to its equivalent JSON format.  
069         * @see java.lang.Object#toString()
070         * @return A string containing the response from the TeleSign server, in JSON format.  
071         */
072        @Override
073        public String toString() {
074                Gson gson = new Gson();
075                return gson.toJson(this);
076        }
077}
